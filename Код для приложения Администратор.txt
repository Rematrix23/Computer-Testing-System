procedure TForm2.LabeledEdit1Change(Sender: TObject);
var
  allq, mark: integer;
begin
  allq:=0;
  tempF:=TIniFile.Create('temp\temp.ini');
  allq:=tempF.Readinteger('settings','all',allq);
  tempF.Free;
  if allq=0 then begin showmessage('Заполните тест'); LabeledEdit1.Text:='0'; end else
  begin
  mark:=strtoint(LabeledEdit1.Text);
  if mark>allq then begin showmessage('Число не должно быть больще количества вопросов'); LabeledEdit1.Text:='0'; end;
  end;
end;
